#import "User.h"

static User *instance = nil;//存储唯一对象

@implementation User

+ (id) shareUser {
    @synchronized(self) {
        if (instance == nil) {
            instance = [[[self class] alloc] init];
        }
    }
    return instance;
}

+ (id) allocWithZone:(struct _NSZone *)zone {
    if (instance == nil) {
        instance = [super allocWithZone:zone];
    }
    return instance;
}

- (id) copyWithZone:(NSZone *)zone {
    return instance;
}

- (id) retain {
    return instance;
}

- (oneway void) release {

}

- (id) autorelease {
    return instance;
}

- (unsigned long) retainCount {
    return UINT_MAX;
}
@end
